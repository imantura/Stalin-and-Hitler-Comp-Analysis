def preprocess_text(text):
    words_remove = ['adolf', 'hitler', 'internet', 'copyright', 'mia', 'markup', 'index', 'volume', 'make',
                    'derivative', 'commercial','source', 'credit', 'source', 'display', 'perform', 
                    'derivative', 'commercial', 'copy', 'languages', 'adolph', 'hitler', 'stalin']
    try:
        text = text.lower()
        text = re.sub(r'[^a-zA-Z\s]', '', text)
        tokens = nltk.word_tokenize(text)
        stop_words = set(stopwords.words('english'))
        tokens = [word for word in tokens if word not in stop_words]
        
        # Lemmatization
        lemmatizer = WordNetLemmatizer()
        tokens = [lemmatizer.lemmatize(word) for word in tokens]
        tokens = [word for word in tokens if len(word) >= 2]
        tokens = [word for word in tokens if word not in words_remove]

        return tokens
    except Exception as e:
        print(f"Error processing text: {e}")
        return []

def compute_avg_similarity(indices1, indices2, cosine_sim_matrix, within_group=True):
    total_sum = 0
    count = 0
    
    for i in indices1:
        for j in indices2:
            if within_group and i == j:  
                continue
            total_sum += cosine_sim_matrix[i, j]
            count += 1
    
    return total_sum / count if count > 0 else 0

# Function to get sentiment and context
def compute_sentiment_with_context(row, lda_model, dictionary):
    # Calculate sentiment score
    text = row['processed_text']
    sentiment_score = analyzer.polarity_scores(text)['compound']
    
    # Get dominant topic and its words
    bow = dictionary.doc2bow(row['tokens'])
    
    # Safely handle empty BoW or missing topics
    if bow:
        topic_probabilities = lda_model.get_document_topics(bow, minimum_probability=0.0)
        if topic_probabilities:
            dominant_topic = max(topic_probabilities, key=lambda x: x[1])[0]  # Get topic with highest probability
            topic_words = lda_model.print_topic(dominant_topic, 10)
        else:
            dominant_topic = None
            topic_words = "No topics found"
    else:
        dominant_topic = None
        topic_words = "No valid words in dictionary"
    
    # Return sentiment, dominant topic, and topic words
    return pd.Series({
        'sentiment': sentiment_score,
        'dominant_topic': dominant_topic,
        'topic_words': topic_words,
        'text_snippet': text[:200]  
    })

def preprocess_tokens(tokens):
    stop_words = set(stopwords.words('english'))

    tokens = [word for word in tokens if len(word) >= 3 and word not in stop_words]
    tokens = [word for word in tokens if not any(remove_word in word for remove_word in words_remove)]
    
    return tokens

def compute_sentiment_with_context(row, lda_model, dictionary):
    # Calculate sentiment score
    text = row['processed_text']
    sentiment_score = analyzer.polarity_scores(text)['compound']
    
    # Get dominant topic and its words
    bow = dictionary.doc2bow(row['tokens'])
    
    # Safely handle empty BoW or missing topics
    if bow:
        topic_probabilities = lda_model.get_document_topics(bow, minimum_probability=0.0)
        if topic_probabilities:
            dominant_topic = max(topic_probabilities, key=lambda x: x[1])[0]  # Get topic with highest probability
            topic_words = lda_model.print_topic(dominant_topic, 10)
        else:
            dominant_topic = None
            topic_words = "No topics found"
    else:
        dominant_topic = None
        topic_words = "No valid words in dictionary"
    
    # Return sentiment, dominant topic, and topic words
    return pd.Series({
        'sentiment': sentiment_score,
        'dominant_topic': dominant_topic,
        'topic_words': topic_words,
        'text_snippet': text[:200]  # Include first 200 characters of the speech as context
    })
